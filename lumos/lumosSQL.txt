BEGIN;
CREATE TABLE "lumos_show" (
    "id" integer NOT NULL PRIMARY KEY,
    "directions_from_parking_to_stage" varchar(200) NOT NULL,
    "parking" varchar(200) NOT NULL,
    "loading_unloading_parking" varchar(128) NOT NULL
)
;
CREATE TABLE "lumos_trick_shows" (
    "id" integer NOT NULL PRIMARY KEY,
    "trick_id" integer NOT NULL,
    "show_id" integer NOT NULL REFERENCES "lumos_show" ("id"),
    UNIQUE ("trick_id", "show_id")
)
;
CREATE TABLE "lumos_trick" (
    "id" integer NOT NULL PRIMARY KEY,
    "audiences" varchar(128) NOT NULL,
    "props" varchar(128) NOT NULL,
    "patter" varchar(1000) NOT NULL,
    "cost_to_preform" decimal NOT NULL,
    "cost_to_buy_effect" decimal NOT NULL
)
;

COMMIT;
